package com.praktikum.mahasiswatest;

/**
 *
 * @author LEGION
 */
public class Mahasiswa {

    private String nama;
    private double nilai;

    public Mahasiswa(String nama, double nilai) {
        this.nama = nama;
        this.nilai = nilai;
    }

    public String getNama() {
        return nama;
    }

    public double getNilai() {
        return nilai;
    }

    public boolean isLulus() {
        return nilai >= 60;
    }
}
